### 方案 1：增强日志输出

```go
func (s *ConfigUpdateService) logNodeTypeStats(index int, nodeLinks []string) {
    tc := make(map[string]int)
    for _, l := range nodeLinks {
        protocol := strings.Split(l, ":")[0]
        tc[protocol]++
    }
    
    // 按协议类型排序输出
    protocols := []string{"vmess", "vless", "trojan", "ss", "ssr", "hysteria", "hysteria2", "tuic"}
    var parts []string
    for _, proto := range protocols {
        if count, ok := tc[proto]; ok && count > 0 {
            parts = append(parts, fmt.Sprintf("%s: %d", proto, count))
        }
    }
    
    // 添加其他未知协议
    for t, c := range tc {
        found := false
        for _, proto := range protocols {
            if t == proto {
                found = true
                break
            }
        }
        if !found {
            parts = append(parts, fmt.Sprintf("%s: %d", t, c))
        }
    }
    
    s.successf("✔️ 节点源 [%d] 下载解析完成: 共提取 %d 个节点 (%s)", 
        index, len(nodeLinks), strings.Join(parts, ", "))
}
```
