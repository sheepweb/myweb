### 步骤2: 创建 i18n 配置

创建 `frontend/src/i18n/index.js`:

```javascript
import { createI18n } from 'vue-i18n'
import zhCN from './locales/zh-CN.json'
import enUS from './locales/en-US.json'
import zhCn from 'element-plus/dist/locale/zh-cn.mjs'
import en from 'element-plus/dist/locale/en.mjs'
import dayjs from 'dayjs'
import 'dayjs/locale/zh-cn'

const messages = {
  'zh-CN': zhCN,
  'en-US': enUS
}

const i18n = createI18n({
  legacy: false,
  locale: localStorage.getItem('user-language') || 'zh-CN',
  fallbackLocale: 'zh-CN',
  messages
})

// Element Plus 语言映射
export const elementPlusLocales = {
  'zh-CN': zhCn,
  'en-US': en
}

// 切换语言
export const setLocale = (locale) => {
  i18n.global.locale.value = locale
  localStorage.setItem('user-language', locale)
  
  // 切换 Element Plus 语言
  const elLocale = elementPlusLocales[locale] || zhCn
  
  // 切换 dayjs 语言
  if (locale === 'zh-CN') {
    dayjs.locale('zh-cn')
  } else {
    dayjs.locale('en')
  }
  
  return elLocale
}

export default i18n
```

## 📈 工作量详细分析

### 需要翻译的内容类型

1. **静态文本**（约60%）
   - 页面标题、按钮文字、标签
   - 工作量：中等，主要是替换工作

2. **动态消息**（约25%）
   - 错误消息、成功提示、验证消息
   - 工作量：较大，需要处理变量插值

3. **表单验证消息**（约10%）
   - 验证规则消息
   - 工作量：较小，统一处理

4. **后端返回消息**（约5%）
   - API错误消息
   - 工作量：较小，后端也需要支持

- **Vue文件**: 41个

## ⚠️ 注意事项

### 1. 后端消息翻译
- 后端返回的错误消息也需要支持多语言
- 建议：后端返回错误码，前端根据语言显示对应消息

### 2. 动态内容
- 用户生成的内容（如用户名、节点名称）不需要翻译
- 日期、数字格式需要根据语言调整

### 3. Element Plus 语言切换
- Element Plus 的语言需要在切换时动态更新
- 可以使用 `ElConfigProvider` 组件包裹应用

### 4. 日期格式化
- dayjs 需要根据语言切换 locale
- 日期显示格式需要适配不同语言
